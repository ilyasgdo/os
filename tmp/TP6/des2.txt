#include <stdlib.h>
#include <unistd.h>
#include <stdio.h>
#include <sys/mman.h>
#include <fcntl.h>
#include <sys/stat.h>
 #include <string.h>
int copieFichier(){
    struct stat st;
    int taille;
    if (stat("./test4.c", &st) == 0) {
        taille= st.st_size; // Taille en octets
    }
    int src =open("./test4.c",O_RDONLY);
    int des = open("./des2.txt",O_CREAT|O_RDWR,0644);
    ftruncate(des,taille*sizeof(char));
    char * tsrc =mmap(NULL,taille*sizeof(char),PROT_READ,MAP_SHARED,src,0);
    char * tdes= mmap(NULL,taille*sizeof(char),PROT_READ|PROT_WRITE,MAP_SHARED,des,0);
    memcpy(tdes,tsrc,taille);
    munmap(tsrc,taille);
    munmap(tdes,taille);
    shm_unlink("./test4.c");
    shm_unlink("./des2.txt");
    return 0;
    
}
int main(){
    copieFichier();
    return 0;
}
